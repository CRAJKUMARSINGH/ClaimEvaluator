// Enhanced Security Headers Implementation app.use((req, res, next) => { if (req.url.startsWith('/api/')) { res.setHeader('X-Content-Type-Options', 'nosniff'); res.setHeader('X-Frame-Options', 'DENY'); res.setHeader('X-XSS-Protection', '1; mode=block'); res.setHeader('X-Powered-By', 'ClaimEvaluator-Secure'); res.setHeader('Strict-Transport-Security', 'max-age=31536000; includeSubDomains'); } next(); }); // Advanced Rate Limiting with Redis Support const rateLimiter = new Map(); const RATE_LIMIT = 200; // requests per minute const RATE_WINDOW = 60 * 1000; // 1 minute function enforceRateLimit(req, res, next) { const clientId = req.ip || req.connection.remoteAddress; const now = Date.now(); const windowStart = now - RATE_WINDOW; if (!rateLimiter.has(clientId)) { rateLimiter.set(clientId, []); } const requests = rateLimiter.get(clientId); const recentRequests = requests.filter(time => time > windowStart); if (recentRequests.length >= RATE_LIMIT) { return res.status(429).json({ error: 'Rate limit exceeded', retryAfter: Math.ceil(RATE_WINDOW / 1000) }); } recentRequests.push(now); rateLimiter.set(clientId, recentRequests); next(); }

// Enhanced Document Processing Pipeline async function processDocument(filePath: string, mimeType: string): Promise { const startTime = Date.now(); try { let content = ''; let metadata = {}; switch (mimeType) { case 'application/pdf': const pdfResult = await processPDF(filePath); content = pdfResult.text; metadata = { pages: pdfResult.numpages, info: pdfResult.info, version: pdfResult.version }; break; case 'application/vnd.openxmlformats-officedocument.wordprocessingml.document': const docxResult = await processWord(filePath); content = docxResult.value; metadata = { messages: docxResult.messages, styleMap: docxResult.styleMap }; break; case 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet': const xlsxResult = await processExcel(filePath); content = xlsxResult.sheets.map(sheet => `Sheet: ${sheet.name}\n${sheet.data.map(row => row.join('\t')).join('\n')}` ).join('\n\n'); metadata = { sheets: xlsxResult.sheets.map(s => s.name), properties: xlsxResult.properties }; break; default: content = await fs.readFile(filePath, 'utf-8'); metadata = { encoding: 'utf-8' }; } const processingTime = Date.now() - startTime; return { content, metadata, processingTime, wordCount: content.split(/\s+/).length, characterCount: content.length, checksum: generateChecksum(content) }; } catch (error) { throw new ProcessingError(`Failed to process document: ${error.message}`, { filePath, mimeType, processingTime: Date.now() - startTime }); } }

// Advanced Financial Calculation Engine class FinancialCalculationEngine { calculateFIDICTraditional(claimData: ClaimData): CalculationResult { const baseAmount = this.calculateBaseAmount(claimData); const timeRelatedCosts = this.calculateTimeRelatedCosts(claimData); const overheadCosts = this.calculateOverheadCosts(claimData); const financingCosts = this.calculateFinancingCosts(claimData); const riskAdjustments = this.calculateRiskAdjustments(claimData); return { baseAmount, timeRelatedCosts, overheadCosts, financingCosts, riskAdjustments, totalAmount: baseAmount + timeRelatedCosts + overheadCosts + financingCosts + riskAdjustments, methodology: 'FIDIC_TRADITIONAL', confidence: this.calculateConfidenceScore(claimData) }; } calculateNHAIHAM(claimData: ClaimData): CalculationResult { const constructionCosts = this.calculateConstructionCosts(claimData); const operationCosts = this.calculateOperationCosts(claimData); const revenueImpact = this.calculateRevenueImpact(claimData); const concessionAdjustments = this.calculateConcessionAdjustments(claimData); const npvCalculation = this.calculateNPV({ cashFlows: [constructionCosts, operationCosts, revenueImpact], discountRate: claimData.discountRate || 0.12, period: claimData.concessionPeriod || 20 }); return { constructionCosts, operationCosts, revenueImpact, concessionAdjustments, npvImpact: npvCalculation, totalAmount: npvCalculation.presentValue, methodology: 'NHAI_HAM', confidence: this.calculateConfidenceScore(claimData) }; } private calculateTimeRelatedCosts(claimData: ClaimData): number { const delayPeriod = claimData.delayMonths || 0; const plantEquipmentCosts = claimData.plantEquipment || 0; const laborCosts = claimData.labor || 0; const materialCosts = claimData.materials || 0; const monthlyCosts = (plantEquipmentCosts + laborCosts + materialCosts) / (claimData.originalDuration || 12); return monthlyCosts * delayPeriod; } private calculateOverheadCosts(claimData: ClaimData): number { const siteOverheads = claimData.siteOverheads || 0; const headOfficeOverheads = claimData.headOfficeOverheads || 0; const delayPeriod = claimData.delayMonths || 0; const monthlyOverheads = (siteOverheads + headOfficeOverheads) / 12; return monthlyOverheads * delayPeriod; } }
